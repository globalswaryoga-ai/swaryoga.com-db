import { useAuth } from '@/hooks/useAuth';
import { useCRM } from '@/hooks/useCRM';
import { useSearch } from '@/hooks/useSearch';
import { usePaginationWithTotal } from '@/hooks/usePagination';

export default function ListPage() {
  const token = useAuth();                              // Auto-redirect if missing
  const crm = useCRM({ token });                        // API calls
  const search = useSearch();                           // Search with debounce
  const pag = usePaginationWithTotal({ pageSize: 20 }); // Pagination

  useEffect(() => {
    crm.fetch(`/api/endpoint?search=${search.query}&skip=${pag.skip}`);
  }, [search.debouncedQuery, pag.page]);

  return (
    <div>
      <input value={search.query} onChange={e => search.setQuery(e.target.value)} />
      <table>{crm.data?.map(item => ...)}</table>
      <button onClick={pag.nextPage}>Next</button>
    </div>
  );
}
